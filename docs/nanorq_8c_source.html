79  uint16_vec KL;

81  kv_resize(uint16_t, KL, N_max);
82  memset(KL.a, 0, kv_size(KL));

85  size_t KL_max =

88  if (KL_max > UINT16_MAX)

96  }

101  }
102

105  for (n = 0; n < kv_size(KL); n++) {
